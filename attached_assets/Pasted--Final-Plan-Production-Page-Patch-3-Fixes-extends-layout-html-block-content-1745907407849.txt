<script>
function planProduction() {
  return {
    recipeId: {{ recipe.id }},
    baseYield: {{ '%.2f'|format(recipe.predicted_yield) }},
    unit: '{{ recipe.predicted_yield_unit }}',
    allowedContainers: [],
    containersSelected: [],
    flexMode: false,
    autoFill: true,
    scale: 1.0,
    stockChecked: false,
    stockCheckPassed: false,
    containmentPercent: 0,
    remainingToContain: 0,

    init() {
      this.fetchContainerPlan();
    },

    get totalContained() {
      return this.containersSelected.reduce((sum, c) => sum + (c.capacity * c.quantity), 0);
    },

    get canStartBatch() {
      const fullyContained = this.flexMode || this.containmentPercent === 100;
      return fullyContained && this.stockChecked && this.stockCheckPassed;
    },

    fetchContainerPlan() {
      fetch(`/api/available-containers/${this.recipeId}?scale=${this.scale}`)
        .then(res => res.json())
        .then(data => {
          this.allowedContainers = data.available;
          if (this.autoFill) {
            this.containersSelected = data.plan;
          }
          this.updateProgress();
        });
    },

    updateProgress() {
      const projected = this.baseYield * this.scale;
      const contained = this.totalContained;
      this.containmentPercent = Math.min((contained / projected) * 100, 100).toFixed(0);
      this.remainingToContain = Math.max(projected - contained, 0);
    },

    manualAddContainer() {
      if (this.autoFill && !this.flexMode) {
        alert('To manually edit containers, please turn off Auto-Fill first.');
        return;
      }
      this.containersSelected.push({ id: '', name: '', capacity: 0, quantity: 1, unit: '' });
      this.updateProgress();
    },

    checkStock() {
      fetch('/api/check-stock', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ recipe_id: this.recipeId, scale: this.scale })
      })
        .then(res => res.json())
        .then(data => {
          this.stockChecked = true;
          this.stockCheckPassed = data.all_ok;
        });
    },

    startBatch() {
      alert('Batch started (connect to real backend).');
    }
  };
}

function containerRow(container, index, parent) {
  return {
    container,
    index,
    allowedContainers: parent.allowedContainers,
    updateContainer() {
      const match = this.allowedContainers.find(c => c.id == this.container.id);
      if (match) {
        this.container.capacity = match.storage_amount;
        this.container.unit = match.storage_unit;
        this.container.name = match.name;

        if (this.container.quantity > match.stock_qty) {
          this.container.quantity = match.stock_qty;
          alert('Cannot allocate more than available container stock.');
        }
      }
      parent.updateProgress();
    },

    removeContainer() {
      parent.containersSelected.splice(this.index, 1);
      parent.updateProgress();
    }
  }
}
</script>
